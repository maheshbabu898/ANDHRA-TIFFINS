  <h2>Andhra Tiffens – Test Order Page</h2>

  <button onclick="placeOrder()">Place Test Order</button>

  <pre id="result"></pre>

  <script>
    function placeOrder() {
      fetch('/order', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          items: ['Idli', 'Masala Dosa'],
          total: 55,
          customer: 'Charan'
        })
      })
      .then(res => res.json())
      .then(data => {
        document.getElementById('result').innerText =
          JSON.stringify(data, null, 2);
      })
      .catch(err => {
        document.getElementById('result').innerText =
          'Error: ' + err;
      });
    }
  </script>
